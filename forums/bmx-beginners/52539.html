Here is something similar I did once. 
Function numbify:String( value:Int )

	Local str:String
	str =  value

	Local num = 0
	Local tmp:String = ""
	
	For Local i = Len(str)-1 To 0 Step -1
		If num = 3 Then
			tmp = "'"+tmp
			num = 1
		Else
			num=num+1
		EndIf

		tmp = Chr(str[i])+tmp
	Next

	Return tmp
EndFunction
