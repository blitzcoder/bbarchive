This will do the trick:
Graphics 640,480,0,0

img:TImage = CreateImage(50,50)

SetImageHandle img,12,45

GetImageHandle img,ix#,iy#

DrawText "Image handle: "+ix+","+iy,0,0
Flip
WaitKey()
End

Function GetImageHandle( image:TImage,x# Var,y# Var)
	x = image.handle_x
	y = image.handle_y
End Function
