TomToad(Posted 2010) [#4]
You can fudge it a bit
Type TBlah
     Field x,y

     Method Do(a:TBlah,b:TBlah=Null)
          If Not b then b = Self
          Self.x = a.x + b.x
          Self.y = a.x + b.x
     End Method
End Type
