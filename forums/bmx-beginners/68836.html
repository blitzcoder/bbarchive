GfK(Posted 2007) [#2]
Global aTypeList:tList = New tList

For N = 1 to 10
  a:aType = new aType
  a.x = rand(0,50)
  a.y = rand(0,20)
  aTypeList.AddLast(a)
Next

For a:aType = EachIn aTypeList
  Print a.x
  Print a.y
Next

Type aType
  Field x:int
  Field y:int
End Type
