	Type tstar
			Field x : Float []
			Field y : Float []
			Field xspeed : Float []
			Field stars : Int ' number of stars
			Method draw()
				For counter =0 To stars-1
					Plot x[counter],y[counter]
				Next'counter
			End Method'draw
			Method move()
				For counter = 0 To stars-1
					x[counter] = x[xcounter] - xspeed[counter]
					If x[counter]<=0 Then x[counter] = 1024
				Next'counter	
			End Method'move
			Method New			
				For counter = 0 To stars-1
					x[counter] = Rand (-1, 1024)
					y[counter] = Rand (-1, 768)

				Next'counter
			End Method'new
 		End Type'Tstjerne
