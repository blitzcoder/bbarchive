Type RGB_Col
Field Red%

tonyg(Posted 2007) [#2]
Type RGB_Col
	Field Red%
	Field Grn%
	Field Blu%
End Type

Global C:rgb_col[6]

For i = 0 To 5
C[i] = New RGB_Col 
Next
