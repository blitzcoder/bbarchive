Strict

Type Ball
Global bl:TList
Field x
Field y

Function Create:Ball(_x:Int, _y:Int)
  Local b:Ball

  b=New Ball
  b.x=_x
  b.y=_y
  If bl=Null Then bl=New TList
  bl.addlast( b )
  Return b
End Function

  For b:Ball=EachIn Ball.bl
  Print "lol"
  Next
End Function
