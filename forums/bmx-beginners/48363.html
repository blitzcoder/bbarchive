Function LoadGlTextures()
Local TextureImage:TPixmap
TextureImage:TPixmap=LoadPixmap("incbin::data\Wall.bmp")

glTexParameteri(GL_TEXTURE_2D,GL_TEXTURE_MIN_FILTER,GL_NEAREST)
