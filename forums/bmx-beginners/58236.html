arrays have sort too!

Local cheese$[10]
Local c:Int=0

cheese[c]="Wasn't";c:+1
cheese[c]="ENAY";c:+1
cheese[c]="the";c:+1
cheese[c]="one";c:+1
cheese[c]="who";c:+1
cheese[c]="likes";c:+1
cheese[c]="cheese?";c:+1

cheese.sort()

DebugLog "case sensitive sorted:"
For c=0 To 9
	DebugLog cheese[c]
Next

End


(run @ debugmode)
