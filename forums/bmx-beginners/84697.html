Function PointsToAngle:Float(x1:Float,y1:Float,x2:Float,y2:Float)
	Local direction#= ATan2(y1-y2,x1-x2)+180
	
	While direction > 360
		direction:-180
	Wend
	
	While direction<0
		direction:+180
	Wend
	
	Return direction

EndFunction

this will return the angle from x,y1 to x,y2.
