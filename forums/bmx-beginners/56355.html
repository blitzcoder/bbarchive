[code]
Graphics 800,600,0 'set gfx mode


Incbin "../gfx/lemmings.jpg"
Global myimage:TImage = LoadAnimImage("../gfx/lemmings.jpg",36,54,0,5)

Global FrameTimer = MilliSecs()
Global frame:Int = 0

EndIf
Flip
Wend
