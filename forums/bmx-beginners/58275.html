Gab, have you got a phobia of testing things out? ;)
Type t1 Abstract
  Global x=5
End Type
Type t2 Extends t1
  Function print_it()
     Print x
  End Function
End Type
Type t3 Extends t1
  Global x=10
  Function print_it()
    Print x
  End Function
End Type
t2.print_it()
t3.print_it()
