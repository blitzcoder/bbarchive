SuperStrict

Type SetType
	Field array:Int[10,10]
End Type

Local maps:settype[10]
foo(maps)

Function foo(n:settype[])

	For Local i% = 0 Until 10
		n[i] = New settype
		For Local y% = 0 Until 10
			For Local x% = 0 Until 10
				n[i].array[y,x] = Rnd(100)
			Next
		Next
	Next
End Function
