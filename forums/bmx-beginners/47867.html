Dreamora(Posted 2005) [#3]
Change it to:
Graphics 800,600
strict
text$=" test....................................................................."
local exit:byte

#mainloop
repeat
  While x <=20000
    Cls
    DrawText text,tickerx#,260
    If MilliSecs() > FrameTimer + 100
      FrameTimer = MilliSecs()
      frame:+1
    EndIf
    x :+ 1
    tickerx=tickerx-5
    Flip
    if keyhit(key_escape)
      exit mainloop
    endif
  Wend
forever
