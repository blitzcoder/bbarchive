EOF(Posted 2008) [#2]
Quick example here

SuperStrict
Framework BRL.GLMax2D

' Setup & initialise variables
Const SW% = 640, SH% = 400
              
' display settings
AppTitle="Move at angle"
SetGraphicsDriver GLMax2DDriver()
Graphics sw,sh,0,40


Global ang#=120.0
Global speed#=1.5

Global x#=20,y#=30

While Not KeyHit(KEY_ESCAPE)
	Cls
	DrawText "Angle="+String(ang),10,10
	x:+Sin(ang)*speed ; y:-Cos(ang)*speed
	DrawRect x,y,20,20
	Flip
Wend
