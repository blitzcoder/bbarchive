Function appendtext:String(filename:String,text:String)
	If FileType(filename) = 0 Then CreateFile(filename)
	Local Stream:TStream = OpenStream(filename)
	SeekStream(Stream, Stream.Size())
	WriteLine(Stream, text)
	CloseStream(Stream)
	FlushMem
End Function
