tonyg(Posted 2007) [#2]
superstrict
graphics 640,480
local a_font:TImagefont=LoadImageFont("<fontpath>/fontname.ttf>",20)
SetImageFont a_font
drawtext "Hello World!",0,0
flip
waitkey()
