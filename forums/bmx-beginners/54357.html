deps(Posted 2005) [#4]
strict

Local dir:String[]
dir = LoadDir( CurrentDir() )

For Local i:Int = 0 Until Len(dir)

	If FileType( dir[i] ) = 2 Then
		' dir[i] is a directory
		Print dir[i]
	ElseIf FileType( dir[i] ) = 1 Then
		' dir[i] is a file
	EndIf
	
Next
