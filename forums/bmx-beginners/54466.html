Type worldobject
    Field x#,y#,z#,name$
    Field target:worldobject
End Type

Global wobs:worldobject[10]

wobs[0]=New worldobject
wobs[0].name$="Door 1"

wobs[1]=New worldobject
wobs[1].name="Switch 1"
wobs[1].target=wobs[0]

Print wobs[1].name+" targets "+wobs[1].target.name
