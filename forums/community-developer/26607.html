B3D codes:
1)
type Sprite
   field Exist
   field X#,Y#
   etc...
end type

2)
Hero.Sprite = new Sprite
Enemy.Sprite = new Sprite

3)
Hero\Exist = true
Enemy\Exist = true

4)
InitSpr(Hero,10,10,etc...)
InitSpr(Enemy,10,10,etc...)

5)
function InitSpr(spr.Sprite,x#,y#,etc...)
   spr\X# = x : spr\Y# = y : spr\etc = ...
end function
