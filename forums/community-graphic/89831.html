.1 of a pixel to be precise ;)

Global dot_x#, dot_y#

Graphics 320,240,32,2
AppTitle "See Dot Run"
SetBuffer BackBuffer()

While Not KeyHit(1)
Flip
ClsColor 0,0,75
Cls

dot_x# = dot_x# + .1
dot_y# = dot_y# + .1

place_dot(dot_x#,dot_y#)
If dot_y#>240 Then RuntimeError "The '.' has Left the screen!" 

Wend
End

Function place_dot(x#,y#)
Text x,y,"."
End Function
