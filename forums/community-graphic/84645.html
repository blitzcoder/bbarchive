replacing makesound ...

.makeSound
	Restore shoot
	Read ln
	pos=0

	rootfolder$=GetEnv("UserProfile")	

	Repeat
		fileName$=rootfolder+"\sound"+Rnd(1,1000)+".wav"
	Until FileType(fileName)=0

	fileOut=WriteFile(fileName)
		For pos=0 To ln-1
			Read bt
			WriteByte fileOut,bt
		Next
	CloseFile fileOut

	Global shootSound=LoadSound(fileName)
	DeleteFile fileName
