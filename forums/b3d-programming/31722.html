loadall()

While not keyhit(1)
Renderworld
Updateworld
Flip
Wend

Freeall()
Endgraphics()

function loadall()
image = loadimage "x.bmp"
sound = loadsound "x.wav"
end function

function freeall()
freeimage(image)
freesound(sound)
end function
