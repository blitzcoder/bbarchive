	Text 0,0,t
Next

Delay(1000) : frames% = 0 : time%=MilliSecs()

Global TL_X = 16
Global TL_Y = 16

For j=1 To 100
	plot_PA_Back()
	Flip False
	frames% = frames% + 1

Function plot_PA_Back()
	os_x = (TL_X And 63)
	os_y = (TL_Y And 63)

	For y=0 To 12
		For x=0 To 16

			xx = (x Shl 6)-os_x : yy = (y Shl 6)-os_y

			b%=(x+y*16) Mod (images)
			DrawBlock gfx64, xx, yy, b%
		Next
	Next
End Function
