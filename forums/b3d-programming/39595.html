Here is one way
Function core()
	; do it here
End Function


Function doMesh(m,xoff#,yoff#,zoff#)

	Local c
	Local childcount=CountChildren(m)
	
	If childcount
		For c=1 To childcount
			core m,xoff#,yoff#,zoff#
			doMesh GetChild(m,c),xoff#,yoff#,zoff#
		Next
	Else
		core m,xoff#,yoff#,zoff#
	EndIf
End Function
