This produces a black light. I'm not quite sure what the orginal effect looked like now...

Graphics3D 800,600
SetBuffer BackBuffer()

light = CreateLight()
LightColor light,-255,-255,-255
TurnEntity light,0,40,0

camera = CreateCamera()
PositionEntity camera,0,0,-2

s = CreateSphere()
EntityColor s,128,128,128


;CameraFogMode camera,1
;CameraFogRange camera,-100,100
;CameraFogColor camera,-255,0,0


While Not KeyHit(1)




	UpdateWorld
	RenderWorld
	Flip
Wend
