Function RColor%(c%)
Return c Shr 16 And 255 Shl 0
End Function

Function GColor%(c%)
Return c Shr 8 And 255 Shl 0
End Function

Function BColor%(c%)
Return c Shr 0 And 255 Shl 0
End Function
