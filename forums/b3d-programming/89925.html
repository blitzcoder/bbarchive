        static private void changeVolume(int chann) {

            double dist = Math.Sqrt( bb.EntityX(CAMERA) * bb.EntityX(CAMERA) + bb.EntityY(CAMERA) * bb.EntityY(CAMERA) + bb.EntityZ(CAMERA) * bb.EntityY(CAMERA));
