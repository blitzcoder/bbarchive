Rob(Posted 2004) [#3]
Here you go:

Graphics 640,480,0,2
SetBuffer BackBuffer()

x=0
y=0
width=0
height=0

;a main loop
While Not KeyHit(1)

	;if the user clicks
	If MouseHit(1)
	
		;first coords
		x=MouseX()
		y=MouseY()
		
		;keep doing this while the user holds the mouse down
		While MouseDown(1)
			width = MouseX()-x
			height = MouseY()-y

			Cls
			Rect x,y,width,height,1			
			
			Flip
		Wend
		
	EndIf
	
Wend
End
