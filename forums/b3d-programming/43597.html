Type window
	Field id%
End Type

Function window_newId%()
	wnd.window = Before Last Window
	If wnd=Null Then Return 0 Else Return wnd\id%+1
End Function

Function window()
	wnd.window = New Window
	wnd\id% = window_newId%()
End Function

While Not KeyHit(1)
	window()
	Delay 2000
Wend
End
