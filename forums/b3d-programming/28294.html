function DropItem(ent, x, z, w, l, p = 1); p = 1 to 100
     For ix = 1 to x + w
         for iz = 1 to z + l
             if rnd(0, 100) < p then
                linepick(ix, 100, iz, 0, -200, 0)
                if pickedentity() <> false then
                     positionentity Ent,ix,pixkedy#(),iz,1
                endif
             endif
         next
     next
end function
