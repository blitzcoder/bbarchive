I have square eyes, I can not see the shadow texture., any suggestions.?



Type ClaseSombras
	Field sombra.Shadow
	Field textura%
End Type 




Function Nuevo_ObjSombra.ClaseSombras( calidadSombra%,fuenteLuz.ClaseLuz )
	
	Local Sombra.ClaseSombras =  New ClaseSombras
	Sombra.ClaseSombras\sombra.Shadow   =      CreateShadow.Shadow(calidadSombra%,1)
	ShadowRange (Sombra.ClaseSombras\sombra.Shadow , 45)
	ShadowPower (Sombra.ClaseSombras\sombra.Shadow, 0.8)
	ShadowLight (Sombra.ClaseSombras\sombra.Shadow,fuenteLuz.ClaseLuz\luz%)	
	Sombra.ClaseSombras\textura%= ShadowTexture(Sombra.ClaseSombras\sombra.Shadow)
	Return Sombra.ClaseSombras
End Function 



PrimerSombra.ClaseSombras = Nuevo_ObjSombra.ClaseSombras(1,luzSol.ClaseLuz)
CreateShadowCaster (PrimerSombra.ClaseSombras\sombra.Shadow,playerMesh.ClasePrimitiva\primitiva)
EntityTexture (planoPlataforma.ClasePrimitiva\primitiva%,PrimerSombra.ClaseSombras\textura%,0,0)
