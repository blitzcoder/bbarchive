file=OpenFile("C:\temp.txt")
if file>0
   type_file=FileType("C:\temp.txt")
  ;you should check if the file exists, it is a directory...
  deletefile("C:\temp.txt")
  control=FileType("C:\temp.txt")
 if control=0 print "File Successfully deleted.."
end if
