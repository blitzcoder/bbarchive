//store a float
BBDECL void BBCALL dllMemStoreFloat(byte* data,int offset,float val) {
byte* buff = data + offset;
memcpy(buff,&val,4);
}

//load a float
BBDECL float BBCALL dllMemLoadFloat(byte* data,int offset) {
byte* buff = data + offset;
float val ;//= 0;
memcpy(&val,buff,4);
return val;
}
