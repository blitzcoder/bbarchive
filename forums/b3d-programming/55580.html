Kev(Posted 2006) [#4]
Type car
	Field x
End Type

Dim cars.car(200)

For loop = 0 To 20
	cars(loop) = New car
	cars(loop)\x = Rnd(5,10)
Next

For loop = 0 To 20
	Print cars(loop)\x
Next

MouseWait
End
