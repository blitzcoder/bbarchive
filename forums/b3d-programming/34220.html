doctor is this what u mean

mesh = LoadAnimMesh("mymodel.b3d")

; grab the anim frames
attack=ExtractAnimSeq(mesh,12,19)

Animate mesh,1,.9,attack,7

Animate mesh,mode,speed,sequnece,tweening
