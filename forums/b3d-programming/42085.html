not very Clean? It's cleaner in my opinion.

Type MyType 
    Field Param1
    Field Param2
End Type

Function MyType_Constructor.MyType()
    Local MyTypeReturn.MyType = New MyType
	MyTypeReturn\Param1 = True
	MyTypeReturn\Param2 = 52
	Return MyTypeReturn
End Function

Local test.MyType = MyType_Constructor()
Print test\Param1
Print test\Param2
