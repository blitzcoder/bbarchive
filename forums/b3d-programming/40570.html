all arrays are global:

Const MAXFOOS = 10
Dim foo(MAXFOOS)

For i = 1 To MAXFOOS
	  foo(i) = Rnd(2,10)	
	Next


Print_foos()


Function print_foos()
	For i = 1 To MAXFOOS
	  Print foo(i)
	Next

End Function
