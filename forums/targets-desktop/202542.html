Okay, this code finally works :)


#include <pwd.h>
char* getPath()
{
char *path = NULL;
int size = 0;

struct passwd *passwd;
passwd = getpwuid(getuid());
size = strlen((passwd->pw_dir));
path = (char *) malloc (sizeof(char) * size);
strcpy(path, passwd->pw_dir);
return path;
}
