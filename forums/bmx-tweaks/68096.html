If you are having problems with keyboard input on Windows, you can always use this:
Function KeyDown:Int(key:Int)
	Select GetAsyncKeyState(key)
		Case 32767,32768,-32767,-32768
			Return True
		Default
			Return False
	EndSelect
EndFunction

Function KeyHit:Int(key:Int)
	Select GetAsyncKeyState(key)
		Case 32767,-32767
			Return True
		Default
			Return False
	EndSelect
EndFunction
