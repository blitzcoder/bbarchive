The class itself should contain a list with all created instances of itself?
Class Coin
    Global all:List<Coin> = New List<Coin>
    
    Field value:Int
        
    Method New()
        value = 0
        all.AddLast(Self)
    End

    Method New(value:Int)
        Self.value = value
        all.AddLast(Self)
    End
End

Function Main:Int()
    Local c1:= New Coin
    Local c2:= New Coin
    Local c3:= New Coin(10)
    
    Print Coin.all.Count
    
    Return 0
End
