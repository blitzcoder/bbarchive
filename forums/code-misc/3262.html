Function Say(Message$,ClearScreen%=True)

 If(ClearScreen = True)
  ClsColor(000,000,000)
  Cls()
  Locate(0,0)
  Color(125,000,250)
 EndIf

 Print(Message)
 Print("Press any key to continue")

 FlushKeys()
 WaitKey()

End Function

Function Ask$(Message$,ClearScreen%=True)

 If(ClearScreen = True)
  ClsColor(000,000,000)
  Cls()
  Locate(0,0)
  Color(125,000,250)
 EndIf

 FlushKeys()
 TAnswer$ = Input$(Message)
 ;DebugLog("TAnswer = "+TAnswer)
 Return TAnswer

End Function
