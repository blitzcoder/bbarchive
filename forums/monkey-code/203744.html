Function CalculateCRC32:String( str:String, byteOrder:Bool = BigEndian, crc:Int = 0 )
crc = crc ~ (-1)

Local length:Int = str.Length

For Local index:Int = 0 Until length Step 1
crc = Lsr(crc, 8) ~ CRC32Table[(crc ~ str[index]) & $FF]
Next

Return DecimalToHex(crc ~ (-1), byteOrder)
End

Const LittleEndian:Bool = False
Const BigEndian:Bool = True

Function DecimalToHex:String( number:String, byteOrder:Bool = BigEndian )
Local charlist:String = "0123456789ABCDEF"
Local char:Int, ret:String, digitOne:Int, digitTwo:Int
Local num:Int = Int(number)

For Local index:Int = 0 Until 4
If byteOrder
char = Lsr(num, 8 * (3 - index)) & $FF
Else
char = Lsr(num, 8 * index) & $FF
End
digitOne = Lsr(char, 4) & 15
digitTwo = char & 15

ret += charlist[digitOne..digitOne + 1] + charlist[digitTwo..digitTwo + 1]
Next

Return ret
End[/monkeycode]
