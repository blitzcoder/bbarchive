	Function FrameSetRow( size:Int[],src:String[],nam:String[])
		
		Local cols = size.length
		Local cs:String
		For Local j=0 Until cols
			If j>0 cs:+","
			cs:+String( size[j] )+"%"
		Next
		
		WriteLine as,"<frameset rows='"+cs+"'>"
		
		For Local j=0 Until cols
			Local txt:String
			txt = "<frame src='"+src[j]+"'"
			If nam[j]<>""
				txt:+" name='"+nam[j]+"'"
			EndIf
			txt:+">"
			WriteLine as,txt
		Next
		WriteLine as,"</frameset>"
			
	End Function
