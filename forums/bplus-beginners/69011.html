Andres(Posted 2007) [#4]
Select ActiveWindow()
    Case ToolWindow
         FreeGadget ToolWindow
    Case AnotherToolWindow
         FreeGadget AnotherToolWindow
    Default
         FreeGadget MainWindow
         End
End Select
