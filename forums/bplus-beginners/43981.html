Function add$(a$,b$)

	l1=Len(a$)
	l2=Len(b$)
	
	If l1>l2 l=l1 Else l=l2

	c=a$
	d=b$
	
	e$=c+d
	
	e$=RSet(e$,l)
	Return Replace(e$," ","0")
	
End Function
