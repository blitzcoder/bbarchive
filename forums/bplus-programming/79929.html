Function ToChar$(txt$,which=0,char$=",")
	Local ow=which
	.beginf
	Local ret$="",sloc=0
	Repeat
		sloc=sloc+1
		If Mid(txt,sloc,1)=char Then 
			If which=0 
				If ret=""
					which=ow+1
					Goto beginf
				Else
					Exit
				EndIf
			Else
				which=which-1
				ret=""
			EndIf
		Else
			ret=ret+Mid(txt,sloc,1)
		EndIf
		If sloc>Len(txt) Return Trim(ret) ;If there's no space, return to the end.
	Forever
	Return (ret)
End Function
