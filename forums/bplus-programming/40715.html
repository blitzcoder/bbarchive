WolRon(Posted 2004) [#5]
;.lib "user32.dll"
;SendMessage%(hWnd%,Msg%,wParam%,lParam%):"SendMessageA"

;Constant related to text area functionality
Const EM_SETREADONLY = $CF

SendMessage QueryObject(YourTextArea, 1), EM_SETREADONLY, True, 0
