;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
; ENCRYPTION FUNCTIONS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;


;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;Enc(ryption)_e(ncrypt)
;description: encrypts a string n$
;
Function Enc_e$(n$)
	l$ = Len(n$)
	For x  = 1 To l$
		t  = Asc(Mid(n$,x,1))
		r$ = r$ +  Chr$( (t-x) - f(x))		
	Next
	Return r$
End Function

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;Enc(ryption)_d(ecrypt)
;description: decrypts an encrypted string n$
;
Function Enc_d$(n$)
	l$ = Len(n$)
	For x  = 1 To l$
		t  = Asc(Mid(n$,x,1))
		r$ = r$ +  Chr$( (t+x) + f(x) )		
	Next
	Return r$
End Function

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;f(n) 
;description: used to aid encryption
;
Function f%(n)
	r = n Shl 1
	r = (r Mod 256)
	Return r%
End Function
