Function ScanDir(path$)
dir = ReadDir(path$)
Repeat
	file$ = NextFile(dir)
	If file$ = "" Then Exit
	If file$ <> "." And file$ <> ".." Then
		Select FileType(path$ + file$)
			Case 1
				Color 0, 255, 0
				Print "File: " + path$ + file$
			Case 2
				Color 255, 0, 0
				Print "Dir: " + path$ + file$ + "\"
				ScanDir(path$ + file$ + "\")
		End Select
	EndIf
Forever
CloseDir dir
End Function
