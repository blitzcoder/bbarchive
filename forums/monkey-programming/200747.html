		Field ScreenSizeX:Int = DeviceWidth()
		Field ScreenSizeY:Int = DeviceHeight()
		Field OffsetX:Int
		Field OffsetY:Int
		
		Method OnCreate()
                OffsetX = ScreenSizeX
                OffsetY = ScreenSizeY
                End
		
		Method ScrollX(x:Float)
			If (x > 0)
				OffsetX = OffsetX + 1
			ElseIf (x<0)
				OffsetX = OffsetX - 1
			End
		End
		
		Method ScrollY(x:Float)
			If (x > 0)
				OffsetY = OffsetY + 1
			ElseIf (x<0)
				OffsetY = OffsetY - 1
			End
		End
