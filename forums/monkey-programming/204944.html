Something like this:
Strict

Function Main:Int()
	Local arrayOfList:List<String>[16]
	For Local i:Int = 0 Until arrayOfList.Length()
		arrayOfList[i] = New List<String>
		If i = 1
			arrayOfList[i].AddLast("1")
			arrayOfList[i].AddLast("2")
		End
	Next
	For Local s:String = Eachin arrayOfList[1]
		Print s
	Next
	Return 0
End
