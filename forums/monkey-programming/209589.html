when a reaches arr.Length
it tries to get to arr[a+3] which is out of array.

maybe check like:
For Local a:Int=0 Until arr.Length Step 5
        if a+3 < arr.Length
	    Local temp:=arr[a]
	    arr[a]=arr[a+3]
	    arr[a+3]=temp
        endif 
Next
