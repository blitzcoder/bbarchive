Strict
Import mojo

Class MyGame Extends App
	Field XMouse:Float
	Field YMouse:Float
	
	Method OnCreate:Int()
		SetUpdateRate 30 ' or 60
		Return 0
	End
	
	Method OnUpdate:Int()
		XMouse = MouseX()
		YMouse = MouseY()
		Return 0
	End
	
	Method OnRender:Int()
		Cls
		SetColor 255,0,0
		DrawRect(XMouse,YMouse,20,40)
		Return 0
	End
End

Function Main:Int()
	New MyGame
	Return 0
End
