Function CodeToString:String(_data:String)
  Local Value:String
    For Local index:Int = 0 To _data.Length
      If _data[index]=10
        Value+="~n"
      Else
        If _data[index]>31 And _data[index]<128
          Value+=String.FromChar(_data[index])
        Endif
      Endif 
    Next  
  Return Value
End Function
