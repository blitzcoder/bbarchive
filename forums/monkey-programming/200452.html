Warpy(Posted 2011) [#5]

Function getDigit(n,i)
	If i=0 Return n Mod 10
	Local e1 = Pow(10,i-1)
	Return ((n Mod (e1*10)) - (n Mod e1))/e1
End

Function Main()
	Local n = 123456789
	
	For Local i=1 To 9
		Print getDigit(n,i)
	Next
End
