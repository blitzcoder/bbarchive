Strict

Function Main:Int()
	Local someString:String = "ABCD"
	Local newString:String
	newString = someString[1..2]
	Print(newString)
	Return 0
End

Here's one way to do it, though it's not pretty!

Function Main ()

	Local test:String = "ABCD"
	
	Print String.FromChar (test [1])
	
End
