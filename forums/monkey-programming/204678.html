Hmmm... try this:

Strict

Import diddy

Function Main:Int()
	' create the array
	Local arr:Int[][] = Arrays<Int>.CreateArray(128,64)
	
	' populate the array
	For Local x:Int = 0 Until 128
		For Local y:Int = 0 Until 64
			arr[x][y] = x + y
		Next
	Next
	
	' print out the array
	For Local x:Int = 0 Until 128
		For Local y:Int = 0 Until 64
			Print arr[x][y]
		Next
	Next
	Return 0
End
