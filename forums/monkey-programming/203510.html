you could for now just add this to the bottom of mojo.graphics
[monkeycode]
Function GetMatrix : Void( m : Float[] )

m[ 3 ] = context.jy
m[ 4 ] = context.tx
m[ 5 ] = context.ty
End Function
[/monkeycode]
