      '@context': 'http://schema.org/',
      '@type': 'ComputerLanguage',
      'name': 'BlitzMax',

     'description': 'BlitzMax is a versatile desktop development environment for game coders with BASIC skills.',
      'image': 'http://www.blitzbasic.com/img/blitzmax_512px.png'

   </script>
