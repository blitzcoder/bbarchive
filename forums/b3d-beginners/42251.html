pexe(Posted 2005) [#8]
graphics 640,480,16,2
apptitle "Getkey"
setbuffer backbuffer()
while not keyhit(28)
k = getkey()
if k
select k
Case 8
If Len(mytext$)>0
mytext$ = Left(mytext$,Len(mytext$)-1)
EndIf
Default
mytext$ = mytext$ + Chr$(k)
end select
k = 0
endif
cls
text 0,0,mytext$
flip
wend
runtimeerror("You typed: "+mytext$)
