Dim pickarray(11)
For a=0 To 11
	pickarray(a)=a

Dim gridarray(2,3)
c=11
For a=0 To 2
	For b=0 To 3
		d=Rand(0,c)
		gridarray(a,b)=pickarray(d)
		pickarray(d)=pickarray(c)
		c=c-1
	Next
Next
