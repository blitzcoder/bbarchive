Type verlet

Function dophysics()
 For v.verlet=Each verlet
  tx#=v\N\x#
  ty#=v\N\y#
  tz#=v\N\z#
  v\N\x#=2*v\N\x#-v\o\x#
  v\N\y#=2*v\N\y#-v\o\y#+gravity
  v\N\z#=2*v\N\z#-v\o\z#
  v\o\x#=tx#
  v\o\y#=ty#
  v\o\z#=tz#
 Next
End Function
