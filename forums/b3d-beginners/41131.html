Rob Farley(Posted 2004) [#3]
or
graphics 1024,768,32,1
setbuffer backbuffer()

bground = loadimage("gfx/bground.bmp")

word$=""

while not keydown(1)
if keyhit(16) then word = word + "q"
drawblock bground
text 0,0,word
flip
wend
