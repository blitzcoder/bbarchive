Beaker(Posted 2004) [#3]
Type alien
Field x,y
End Type

For f= 0 To 10
a.alien = New alien
If f = 10 a\x = 20
Next

a = Last alien
Print a\x
Stop
End
OR
Type alien
Field x,y
End Type

For f= 0 To 10
a.alien = New alien
If f = 10 a\x = 20
Next

a = First alien
While a <> Last alien
a = After a
Wend

Print a\x
Stop
End
