The inlcude file
; Locate/Print for Blitz+
Global locateX,locateY

Function Locate(x,y)
	locateX=(x-1)*StringWidth("W") : locateY=(y-1)*FontHeight()
End Function

Function Print(txt$)
	Text locateX,locateY,txt$
End Function
