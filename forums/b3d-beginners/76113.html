chi(Posted 2008) [#2]
;;;;; user32.decls ;;;;;
; .lib "user32.dll"
;
; api_HasFocus%() : "GetActiveWindow"
; api_IsIconic% (hwnd%) : "IsIconic"
;;;;;;;;;;;;;;;;;;;;;;;;

Global hwnd=SystemProperty("apphwnd")
Graphics3D 800,600,0,2

Repeat
	Cls
	RenderWorld

	If api_HasFocus()<>hwnd
		txt$="window lost focus"
	Else
		txt$="window has focus"
	EndIf
	
	If api_IsIconic(hwnd)
		AppTitle"minimized"
	Else
		AppTitle"windowed"
	EndIf

	Text 10,10,txt$
	Flip
Until KeyHit(1)
End
