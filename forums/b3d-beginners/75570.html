Function timer()
Locate 0,0

hours=Input("hours? ")

secs=Input("seconds? ")

WaitTimer milsec
secs=secs-1

mins=mins-1
Else If mins=-1
mins=0

hours=0
PlaySound (beep)
End If

mins=60
hours=hours-1
End If

mins=0
secs=0
hours=0
PlaySound beep
End If

End

;************************************************************************************************
Function clock()
gettime()
While Not KeyHit(1)
Locate 0,0
Cls
Text 40,0,":"+secs,0,0
Text 15,0,":"+mins,0,0
Text 0,0,hours,0,0
WaitTimer milsec
secs=secs+1
If secs>60
secs=0
mins=mins+1
End If
If mins>60
mins=0
hours=hours+1
End If
Wend
End
End Function

Function gettime()
time$=CurrentTime()
hours=Left$(time$,2)
mins=Mid$(time$,4,2)
secs=Right$(time$,2)
End Function
