Happy Sammy(Posted 2008) [#2]
extracted from help file of "LoadTexture"
cube=CreateCube() 
PositionEntity cube,0,0,5 

; Load texture 
tex=LoadTexture( "media/b3dlogo.jpg" ) 

; Texture cube with texture 
EntityTexture cube,tex

Beaker(Posted 2008) [#4]
