Instead try this:
vh_fadeintro_sfx = LoadSound ("audio/fade-intro.wav")
If  vh_fadeintro_sfx <> false Then
 SoundVolume (vh_fadeintro_sfx,maxvolume)
 PlaySound (vh_fadeintro_sfx,infinite)
