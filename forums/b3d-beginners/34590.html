Function MakeWord( hi, low )
	tempbank = CreateBank(2)
	PokeByte tempbank,0,hi
	PokeByte tempbank,1,low
	tempshort = PeekShort(tempbank,0)
	FreeBank tempbank
	Return tempshort
End Function
