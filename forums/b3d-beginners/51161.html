Matty(Posted 2005) [#3]
sumx#=0
sumz#=0
sumy#=0
vcount=0
for s=1 to countsurfaces(mesh)
surf=getsurface(mesh,s)
for v=0 to countvertices(surf)-1
sumx#=sumx#+vertexx(surf,v)
sumy#=sumy#+vertexy(surf,v)
sumz#=sumz#+vertexz(surf,v)
vcount=vcount+1
next

next
if vcount>0 then 
avgx#=sumx/vcount
avgy#=sumy/vcount
avgz#=sumz/vcount
positionmesh mesh,-avgx,-avgy,-avgz
endif
