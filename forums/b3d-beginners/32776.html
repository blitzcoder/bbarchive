Yan(Posted 2004) [#2]
Type fred
  Field ears
End Type

Dim bloke.fred(9)

For n=0 to 9
  bloke(n) = new fred
  bloke(n)\ears = Rand(4)
Next

show_fred()

Waitkey()

end


Function show_fred(); Arrays are global
  for n=0 to 9
    print "Fred now has " + bloke(n)\ears + " ears!"
  Next
End Function
