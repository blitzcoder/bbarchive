GfK(Posted 2006) [#3]
Try this:
Graphics3D 800,600
zoom# = 1
cam = CreateCamera()
cube = CreateCube()
light = CreateLight()
MoveEntity cam,0,5,-50

While Not KeyDown(1)
	If KeyDown(200) Then zoom = zoom + 0.1
	If KeyDown(208) Then zoom = zoom - 0.1
	If zoom < 1 Then zoom = 1
	CameraZoom cam,zoom
	
	RenderWorld
	Text 0,20,"Cam Zoom: " + zoom
	Text 0,10,"Field of view: " + FieldofView(zoom)
	Flip
Wend

Function FieldOfView#(zoom#)
	Local fov#
	fov = 2 * ATan(1 / zoom)
	Return fov
End Function
