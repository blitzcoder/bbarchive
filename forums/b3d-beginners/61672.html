If PickedEntity() > 0 Then
temprot# = 0
If PickedNZ() > 0 Then temprot# = PickedNZ() * -180

RotateEntity gunsprite, PickedNY() * 90, (PickedNX() * 90) + temprot#, 0

EntityAlpha gunsprite, 1
Else
EntityAlpha gunsprite, 0
EndIf
