[edit] Yep - that did it - thanks again!

In case it helps anyone having a similar problem, the working code looks like this:
wndMain_mnuFile=CreateMenu("File",100,WindowMenu(wndMain:TGadget))
wndMain_mnuNew=CreateMenu("New",101,wndMain_mnuFile:TGadget,KEY_N,MODIFIER_CONTROL)
wndMain_mnuLoad=CreateMenu("Load",102,wndMain_mnuFile:TGadget,KEY_L,MODIFIER_CONTROL)
wndMain_mnuSaveAs=CreateMenu("Save As...",103,wndMain_mnuFile:TGadget,KEY_S,MODIFIER_CONTROL)
CreateMenu("",0,wndMain_mnuFile)
wndMain_mnuQuit=CreateMenu("Quit",104,wndMain_mnuFile:TGadget,KEY_Q,MODIFIER_CONTROL)
