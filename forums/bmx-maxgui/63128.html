kfprimm(Posted 2006) [#2]
replace,
CreateMenu"&New",MENU_NEW,filemenu,KEY_N,MODIFIER_COMMAND

with,
Local newmenu:TGadget=CreateMenu("&New",MENU_NEW,filemenu,KEY_N,MODIFIER_COMMAND)
CreateMenu"New Project",MENU_NEWPROJECT,newmenu
CreateMenu"New File",MENU_NEWFILE,newmenu
