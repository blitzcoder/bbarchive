I'd just build a function for this, what is what you suggested anyways.

Function MouseZSpeed:Int() 
	Global _mz:Int
	Local mz:Int = MouseZ()
	If mz = _mz Then
		Return 0
	EndIf
	If mz > _mz Then
		_mz = mz
		Return 1
	EndIf
	_mz = mz
	Return -1

End Function
