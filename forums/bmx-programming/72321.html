Graphics 800 , 600

color2 = $ff00ff00

color4 = $ffff00ff

D3D7Max2DDriver().vrts[7] = color2
D3D7Max2DDriver().vrts[11] = color3
D3D7Max2DDriver().vrts[15] = color4
DrawRect 0 , 0 , 100 , 100
Flip
