Aha - these seem to work:
Extern
	Function lua_pushblitzobject(_st:Byte Ptr, obj:Object) = "lua_pushlightuserdata"
	Function lua_toblitzobject:Object(_st:Byte Ptr,index:Int) = "lua_topointer"
End Extern
