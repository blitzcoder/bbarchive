EDIT:
This is good enough (for what I needed to do):
Function RemoveEmptyArrayValues:String[] (in:String[])
  Local out:String[], olen:Int = 0
	
	For Local i:Int = 0 To in.Length - 1
	  Local value:String = in[i]
		
		If value <> Null
		 olen:+ 1
			
			out = out[..olen]
			out[olen - 1] = value
			
		End If
		
	Next
	
   Return out
   
End Function
