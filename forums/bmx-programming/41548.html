' A hello type
Type hello
	Field x
End type

' Make a tlist to hold all hellos
list:tlist = New tlist

' Make 6 hellos and add them to the list
For i = 0 To 5
	h:hello = New hello
	h.x = Rand(0,1000)
	List.AddLast h
next

' Go through the list and print to the output window
For h:hello = EachIn list
	Print h.x
next
