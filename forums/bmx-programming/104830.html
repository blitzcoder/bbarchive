					(DeskTopSize.CX-WindowSize.CX)/2 ,..
					(DeskTopSize.CY-WindowSize.CY)/2 ,..
					WindowSize.CX ,..
					WindowSize.CY ,..
					Null,..
					WINDOW_CLIENTCOORDS)
													 
Global Pane:TGadget	= CreateCanvas(	0,..
					0,..
					WindowSize.CX,..
					WindowSize.CY,..
					MainWindow)
