Brucey(Posted 2009) [#2]
Surprisingly, it works just like any other cast :
SuperStrict

Function myFunction(arr:Object)
	Local array:Int[] = Int[](arr)
	
	Print array.length
	Print array[1]
End Function

Local myArr:Int[2]
myArr[0] = 12
myArr[1] = 13

myFunction(myArr)
