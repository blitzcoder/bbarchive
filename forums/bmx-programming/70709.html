Well, here we are:
Type FuncPointerType
  Field Callback(foo%)

  Method Trigger()
     If Callback<>Null Then Callback(42)
  End Method
End Type

Function MyFunction(i%)
 Print "The answer is "+i
End Function

Global test:FuncPointerType=New FuncPointerType
test.Callback=MyFunction
test.Trigger()
