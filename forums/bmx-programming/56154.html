Function showlists()
	Print "---------------"; pos1=0
	For main:TMain=EachIn list; pos1:+1
		Print "main "+pos1; pos2=0
		For sub:TSub=EachIn main.SubList; pos2:+1
			Print "~tsub "+pos2; pos3=0
			For extra:TExtra=EachIn sub.ExtraList; pos3:+1
				Print "~t~textra "+pos3+" =  X:"+extra.X+"  Y:"+extra.Y
			Next
		Next
	Next; Print "---------------"
EndFunction
