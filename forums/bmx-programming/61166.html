Brien,
Try this:

' return -1 to -180, or 1 to 180
Function TurnToFace:Int(A1#, A2#)

	Local ret:Int
	
	ret = A1-A2
	If ret >= 180
		ret = -(360 - ret)
	Else
		If ret <= -180
			ret = ret + 360
		EndIf
	EndIf

    Return ret

End Function
