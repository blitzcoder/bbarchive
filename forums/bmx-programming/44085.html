Here are two ways to convert a signed short to an Int.
' Human-readable version:

Function SignedShortToInt( s:Short )
	
	If s < 32768 Then Return Int(s)
	Return Int(s) - 65536
	
End Function


' CPU-friendly version:

Function SignedShortToInt( s:Short )
	
	Return (s Shl 16) Sar 16
		
End Function
