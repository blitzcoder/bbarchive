	Field gun:TGun
	Method New()
		Self.gun = New TGun
	EndMethod

Type TGun
	Field bullets:TBullets
	Method New()
		Self.bullets = New TBullets
	EndMethod

	Method Shoot()
		If Self.bullets.count()>0
			Self.bullets.add -1
			Return True
		EndIf
		Return False
	End Method
	
EndType
Type TBullets
	Field amount:Int
	
	Method Add(count:Int)
		Self.amount:+count
	EndMethod
	
	Method Count()

	EndMethod
