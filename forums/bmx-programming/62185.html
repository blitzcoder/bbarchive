Local tokens$[2]

tokens[1]="search string 2"

ScanPath("ing 2",tokens)

Function ScanPath(path$,tokens$[]) 
	
	For Local s:String = EachIn tokens
		If s.find(path)=>0
			Print "~q"+s+"~q contains ~q"+path+"~q"
		EndIf
	Next
	
EndFunction
