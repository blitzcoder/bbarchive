Ahh...Right, then perhaps something this would be okay?...
Function DrawTextAll(str$, x, y)
  If TD3D7Max2DDriver(_max2dDriver)
    DrawText str$, x, y
  else
    GLDrawtext str$, x, y
  endif
end function
