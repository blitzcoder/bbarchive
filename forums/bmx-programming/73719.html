Type Test1
End Type

Type Test2 Extends Test1
End Type

Local t2:Test2

Local t1:Test1

t2 = New Test2
t1 = New Test1

If Test2(t1) Then
Print " Same Type "
Else
Print "Not Same Type"
End If


If Test2(t2) Then
Print " Same Type "
Else
Print "Not Same Type"
End If
