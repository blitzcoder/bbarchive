A parallax scrolling starfield.\ESC - Quit

244 bytes.

w=320;h=240;Graphics w,h
d=0
While Not KeyHit(27)
	Cls
	SeedRnd(1)
	For s=0 To 200
		x=Rnd(0,w)
		If s > 100 m=2 Else m=1
		y=Rnd(0,h)+d*m
		If y > h Then y:-h
		Plot x,y
		If m=2 Plot x,y-h
	Next
d:+1
If d>h Then d=0
	Flip
Wend

g=300
Graphics g,g,0 'square graphics window
