jsp(Posted 2011) [#3]
Do you mean to store them for your own reference for fast access?
Type TSomething
	Global list:TList = New TList
	Field _link:TLink

	Method New()
		_link = list.AddLast(Self)
	End Method

	Function Create: TSomething ()
		Local S: TSomething = New TSomething 
		Return S
	End Function
	
	Method Kill()
		_link.Remove()
	End Method

End Type
