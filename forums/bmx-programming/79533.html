Graphics 400,300, 0

Cls
SetBlend(lightblend)

SetRotation(-15)

DrawLine(50,50,80,300,False)
For Local i:Int = 1 To 20
	SetLineWidth(i+1)
	SetAlpha(0.05)
	DrawLine(50,50,80,300,False)
Next

SetRotation(-45)

DrawLine(150,50,180,300,False)
For Local i:Int = 1 To 20
	SetLineWidth(i+1)
	SetAlpha(0.05)
	DrawLine(150,50,180,300,False)
Next

Flip
