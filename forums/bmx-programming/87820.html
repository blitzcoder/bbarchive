Function Convex(p:XY[],n:Int)

	Local i:Int,j:Int,k:Int
	Local flag:Int = 0;
	Local z:Double
	
	If (n < 3) Return 0
		
	For i=0 Until n
		j = (i + 1) Mod n
		k = (i + 2) Mod n
		z = (p[j].x - p[i].x) * (p[k].y - p[j].y)
		z :- (p[j].y - p[i].y) * (p[k].x - p[j].x)
		If (z < 0)
			flag :| 1;
		Else 
			If (z > 0) flag :| 2
		EndIf
		If (flag = 3) Return(CONCAVE);
	Next
	If (flag <> 0)Return(CONVEX) Else Return 0
End Function
