fredborg(Posted 2008) [#9]
SuperStrict

Local y:Float

Local a:Int = MilliSecs()
For Local i:Float = 0.0 To 1000000.0 Step 1.0
	y = Sqr(i)
Next
a = MilliSecs()-a

Local b:Int = MilliSecs()
For Local i:Float = 0.0 To 1000000.0 Step 1.0
	y = i^0.5
Next
b = MilliSecs()-b

Print a+" vs "+b

Local arr:Float[1000000]

	arr[i] = Sqr(i)
Next

Local c:Int = MilliSecs()

	y = arr[i]
