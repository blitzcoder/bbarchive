It gets worse, that only converts a single character. You have to do this

void stoupper(std::string& s)
	{
	  std::string::iterator i = s.begin();
	  std::string::iterator end = s.end();
	 
	  while (i != end) {
	    *i = std::toupper((unsigned char)*i);
	    ++i;
	  }
	}
