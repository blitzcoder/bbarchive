Function FBK_CryptString$(Source$,Key$)
	Local ls = Len(Source$)
	Local lk = Len(Key)
	For C=1 To ls
		Char$=Char$+Chr$(Asc(Mid$(Source$,C,1) ) ~ Asc(Mid$(Key$,1 + (C Mod lk),1) ))
	Next

	Return Char$
End Function

Function xor_crypt:String( msg$, pass$ )
	Local ml = Len(msg$)
	Local pl = Len(pass$)
	Local result$

	For Local i = 0 Until ml
		Local c:Byte = msg$[i]
		Local k = pass$[ i Mod pl ]
	
		c = c ~ k
	
		result$:+Chr(c)
	Next 

	Return result$
EndFunction
