Function InsertAfter(list:TList,a:Object,b:Object)
	Local	link:TLink
	link=list.FindLink(a)
	list.InsertAfterLink(b,link)
End Function

Function InsertBefore((list:TList,a:Object,b:Object)
	Local	link:TLink
	link=list.FindLink(a)
	list.InsertBeforeLink(b,link)
End Function

t:TList=New TList
t.AddLast "A"
t.AddLast "B"

	Print a
