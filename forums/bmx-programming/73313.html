This works - but it uses MaxGUI.

Local SW:Int = 800
Local SH:Int = 600

MyWindow:TGadget=CreateWindow(AppTitle, (GadgetWidth(Desktop())-SW)/2,(GadgetHeight(Desktop())-SH)/2,SW,SH,Null,WINDOW_CLIENTCOORDS|WINDOW_TITLEBAR)
