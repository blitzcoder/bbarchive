Function Main:Int()
	Local thread1:C_App

	Local thread4:C_App
	
	
	thread1 = New C_App("1'st thread")

	thread3 = New C_App("3'rd thread")
	thread4 = New C_App("Hey Freddy!, look at all them Freds")
	
	Return 0
	
End




Class C_App extends App
	Field message:String
	
	
	Method OnCreate:Int()
		
		Return 0

		Return 0
		
	End
	
	
	Method New(message:String)

	End
	
End
