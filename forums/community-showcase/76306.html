Local LastX=0;

' 1.33 is the 4x3 format. 1.25 can been used for 5x4 format, if you want to add these as well.
For Local mode:TGraphicsMode=EachIn GraphicsModes()
	If STR_Round(Float(mode.width)/Float(mode.height),2)="1.33" And mode.depth=32
		If mode.width<>LastX And mode.height<>LastY
			LastX=Mode.Width
			LastY=Mode.Height
			Print "resolution found: "+mode.width+" "+mode.height
		EndIf
	EndIf
Next


Function STR_Round:String(value:Double,digit:Int)
	Local result:String
	Local data:String = String(value)
	Local on=False
	For Local i=1 To Len(data)
		If on=1 Then digit:-1
		If digit=0 Then result=Left$(data,i); Exit;
		If Mid$(data,i,1)="." Then on=1
	Next

	For i=1 To 32
		If Right(result,1)="0" Then result=Left$(result,Len(result)-1)
	Next
	If Right(result,1)="." Then result=Left$(result,Len(result)-1)
	Return result
End Function
