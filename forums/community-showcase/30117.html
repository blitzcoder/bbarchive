Function AppIcon(icon$)

	;------------------------------------------------------------------
	;set the current application icon (the little one in the title bar)
	;------------------------------------------------------------------

	hWND=User32_GetActiveWindow()
	icon=LoadIcon(hwnd,icon,0)
	User32_SetClassLong hwnd,-14,icon

End Function
