I always intended to add at least 20 game modes and 100 challenges which will be variations of the game modes ( timed or with a set number of lives ). I hope to b able to include the ability to create your own games, the process of adding one is quite simple .. e.g. this is Survival in it's entirety, hard coded atm but will be a mini-game file eventually :

Case "SURVIVAL"
		
			LEVEL\Weapons = 1
			LEVEL\SelfDestructDamage = 0
			LEVEL\SelfDestruction = 1
			LEVEL\Lives = FIRST_TO
			LEVEL\Possession = 0
			LEVEL\Damage = 1
			LEVEL\Scoring = LOSE_LIFE
			LEVEL\ScoreTarget = 0
			LEVEL\PointerVisible = 0
			LEVEL\RespawnShield = 1
			LEVEL\ShieldAtStart = 1
									
			For l = 0 To TEAMS-1
				VEHICLEcreate( "TRUCK", l, -1 , TEAM_MULTI , LEVEL\Lives, "FRIGHTENED" )
			Next

			VEHICLEcreate( "TRUCK" , 8 , -1,2 , -1, "AGGRESSIVE" )
			VEHICLEcreate( "HUMVEE" , 8 , -1,2 , -1, "AGGRESSIVE" )
			VEHICLEcreate( "TANK" , 8 , -1,15 , -1, "AGGRESSIVE" )
			VEHICLEcreate( "GUNSHIP" , 8 , -1,3 ,-1, "AGGRESSIVE" )
			VEHICLEcreate( "STINGER", 8, -1,3,-1, "AGGRESSIVE" )
			VEHICLEcreate( "SENTRY" , 8 , -1,3 ,-1, "AGGRESSIVE" )
			BODYcreate( "MINE" , 8 , 15 )
			BODYcreate( "SUPERMINE" , 8 , 5 )
