it's possible :) you must only create some fonction

Function MainWindows()
Main=BCF_CreateWindow(....)
; Create Button etc...

repeat
BCF_RenderBCF()
forever

BCF_CloseWindow(Main)
Options()
end function

Function Options()
Options=BCF_CreateWindow(....)
; Create button etc ...

repeat
BCF_RenderBCF()
forever

BCF_CloseWindow(Options)
MainWindows()
End function
