I'm making something to do with rockets.

AppTitle=""
Graphics 600,600
SetBlend ALPHABLEND

off#=0
While Not (KeyHit(KEY_ESCAPE) Or AppTerminate())

	diag#=Sqr(2*300*300)
	rad#=Sqr(diag*diag-200*200)
	off:+.02
	SetAlpha .12
	a#=0
	While a<360
		an#=a+off
		SetColor 0,(Cos(a*3+off*10)*.5+.5)*255,(Sin(a*2-off)*.5+.5)*255
		x#=300+Cos(an)*200
		y#=300+Sin(an)*200
		DrawLine x,y,x+Cos(an+90)*rad,y+Sin(an+90)*rad
		DrawLine 300-Cos(an)*160,300-Sin(an)*160,300-Cos(an)*120,300-Sin(an)*120
		DrawLine 300+Cos(an)*80,300+Sin(an)*80,300+Cos(an)*40,300+Sin(an)*40
		a:+.2
	Wend
	
	Flip
	Cls
Wend
