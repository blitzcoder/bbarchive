Beaker(Posted 2008) [#9]
Graphics3D 800,600
SetBuffer BackBuffer()

c = CreateCamera()
CameraProjMode c,2
CameraZoom c,0.7
MoveEntity c,0,0,-5.5

b = LoadMesh("dwarf4.b3d")
;b = LoadMesh("arobot.b3d")
TurnEntity b,0,180,0

FitMesh b,-1,-1,-1,2,2,2,True

box = CreateCube()
EntityAlpha box,0.3

While Not KeyDown(1)

	RenderWorld
	Flip
	TurnEntity b,0,0.1,0
	Delay 3	
Wend
End
