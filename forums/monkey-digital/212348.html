	Print "Hours   = " + time.Hours
	Print "Day     = " + time.Day
	Print "WeekDay = " + time.WeekDay
	Print "YearDay = " + time.YearDay
	Print "Month   = " + time.Month
	Print "Year    = " + time.Year
	Print "time    = " + time.ToString()
End
