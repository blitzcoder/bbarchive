yes you have to set two commandos before you draw something non-3d!

Repeat
	RenderWorld()
	BeginMax2D

		DrawImage AnyImage,0,0

		SetImageFont font
		SetColor 50,150,250
		DrawText "Hallo",0,0
	EndMax2D

	Flip 0
	WaitTimer fps
Until KeyHit(KEY_ESCAPE)
