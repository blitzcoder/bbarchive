We recently added getBrowserName in Diddy, so you can do something like this:
Import diddy

Function Main:Int()
	Local browser:=GetBrowserName()
	Local sndfrmt:="ogg"
	
	Print browser
	
	Select browser
		Case "Chrome"
			sndfrmt= "ogg"
		Case "Firefox"
			sndfrmt= "ogg"
		Case "Explorer"
			sndfrmt= "m4a"
	End
	Print "using "+sndfrmt
	
	Local sndTest:Sound = LoadSound("test"+sndfrmt)
	
	Return 0
End
